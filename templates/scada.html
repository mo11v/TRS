① Arduino Sketch
// TRS Platform — ESP32 Client // Libraries: WiFi, HTTPClient, ArduinoJson #include <WiFi.h> #include <HTTPClient.h> #include <ArduinoJson.h> // ── CONFIG (عدّل هنا) ────────────────────── const char* ssid = "YOUR_WIFI"; const char* wifiPass = "YOUR_PASS"; const char* serverUrl = "http://192.168.1.100:8000/api/esp/push"; const char* apiKey = "trs-esp-key-2026"; const char* devCode = "TRS-001"; // Device code في TRS int jobId = 1; // Job ID في TRS int interval = 1000; // ms بين كل reading // ── PINS ─────────────────────────────────── #define TORQUE_PIN 34 // Analog — load cell / strain gauge ADC #define RPM_PIN 35 // Analog — hall sensor // ── CALIBRATION ──────────────────────────── float torqueScale = 0.5; // ADC raw → ft·lb float rpmScale = 0.01; // ADC raw → RPM int jointNum = 0; float turns = 0.0; float prevTorque = 0.0; void setup() { Serial.begin(115200); WiFi.begin(ssid, wifiPass); Serial.print("WiFi..."); while (WiFi.status() != WL_CONNECTED) { delay(400); Serial.print("."); } Serial.println("\nIP: " + WiFi.localIP().toString()); } void loop() { if (WiFi.status() != WL_CONNECTED) { WiFi.reconnect(); delay(2000); return; } // ── Sensors ───────────────────────────── int rawT = analogRead(TORQUE_PIN); float torque = rawT * torqueScale; float rpm = analogRead(RPM_PIN) * rpmScale; turns += rpm / 60.0 * (interval / 1000.0); // Detect new joint (torque drops after makeup) if (prevTorque > 100 && torque < 20) { jointNum++; turns = 0.0; Serial.printf("★ New joint: %d\n", jointNum); } prevTorque = torque; // ── Send to TRS ───────────────────────── HTTPClient http; http.begin(serverUrl); http.addHeader("Content-Type", "application/json"); http.addHeader("X-ESP-Key", apiKey); StaticJsonDocument<256> doc; doc["device_code"] = devCode; doc["job_id"] = jobId; doc["torque"] = torque; doc["turns"] = turns; doc["rpm"] = rpm; doc["joint_num"] = jointNum; String body; serializeJson(doc, body); int code = http.POST(body); Serial.printf("→ [%d] T=%.1f RPM=%.1f J=%d\n", code, torque, rpm, jointNum); http.end(); delay(interval); }
